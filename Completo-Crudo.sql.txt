DECLARE @PromedioOriginal MONEY;
DECLARE @PrecioMinimoModificado MONEY;
DECLARE @FilasModificadas INT;

BEGIN TRY

-- 1. Iniciar transación 
BEGIN TRANSACTION;
PRINT '>>> Iniciando proceso de actualización nativa';

-- Obtener promedio original
SELECT @PromedioOriginal = AVG(ListPrice)
FROM Production.Product
WHERE ListPrice > 0;

PRINT 'Auditoria: Promedio original de precios:$' + CAST(@PromedioOriginal as VARCHAR);

-- 2. Logica de negocios:

    BEGIN TRANSACTION;
    PRINT '>>> Transacción iniciada';

    -- Actualización masiva
    UPDATE Production.Product
    SET ListPrice = ListPrice * 1.15
    WHERE ListPrice > 0;

    SET @FilasModificadas = @@ROWCOUNT;
    PRINT '>>> Auditoria: Productos afectados por el incremento: ' + CAST(@FilasModificadas AS VARCHAR(10));

-- 3. Control de transacciones.
    SELECT @MinimoPostIncremento = MIN(ListPrice)
    FROM Production.Product
    WHERE ListPrice > 0;

    PRINT '>>> Auditoria: El precio mínimo post-incremento es: $' + CAST(@MinimoPostIncremento AS VARCHAR(20));

    IF @MinimoPostIncremento <= @PromedioOriginal
    BEGIN
        PRINT '>>> Alerta: El mínimo no supera el promedio. Se debe revertir.';
        ROLLBACK TRANSACTION;
        PRINT '>>> Auditoria: Transacción revertida (ROLLBACK).';
    END
    ELSE
    BEGIN
        COMMIT TRANSACTION;
        PRINT '>>> Auditoria: Transacción confirmada (COMMIT).';
    END

END TRY
BEGIN CATCH

-- 4. Manejo de errores dinamicos:

    PRINT '>>> Error detectado: ' + ERROR_MESSAGE();
    
    IF @@TRANCOUNT > 0
    BEGIN
        ROLLBACK TRANSACTION;
        PRINT '>>> Auditoria: Se detectó transacción abierta y fue revertida.';
    END

    -- Relanzar el error
    DECLARE @Msg NVARCHAR(2048) = 'Error en el proceso: ' + ERROR_MESSAGE();
    THROW 50000, @Msg, 1;
END CATCH;

-- Tabla A: Socios.FichaPersonal (Ubicada físicamente en PRIMARY)

    SocioID INT IDENTITY(1,1),
    Documento Socios.TipoDocumento,     -- Uso del UDT Obligatorio
    Nombre VARCHAR(50) NOT NULL,
    Apellido VARCHAR(50) NOT NULL,

    CostoMensual DECIMAL(10,2) NOT NULL,

) ON [PRIMARY];
GO


-- Tabla C: Socios.RegistroAccesos (Ubicada explícitamente en HISTORICO)
CREATE TABLE Socios.RegistroAccesos (
    AccesoID BIGINT IDENTITY(1,1),
    SocioID INT NOT NULL,
    FechaHoraAcceso DATETIME NOT NULL DEFAULT GETDATE(),
    NroMolinete INT NOT NULL,
    Permitido BIT NOT NULL, -- 1: Sí, 0: No
    
    -- Restricción de Clave Primaria Compleja
    CONSTRAINT PK_RegistroAccesos PRIMARY KEY CLUSTERED (AccesoID, FechaHoraAcceso),
    
    -- Restricción de Clave Foránea (Integridad Referencial)
    CONSTRAINT FK_RegistroAccesos_FichaPersonal FOREIGN KEY (SocioID)
        REFERENCES Socios.FichaPersonal (SocioID)
) ON [HISTORICO]; -- Asignación explícita al Filegroup HISTORICO

PRINT '';

PRINT '>>> FASE 4: Auditoría de Distribución de Registros en el Motor';
PRINT '======================================================================';

-- Consulta metadata avanzada para certificar la distribución en almacenamiento
SELECT 
    p.partition_number AS NroParticion,
    fg.name AS NombreFilegroup,
    p.rows AS CantidadDeRegistros,
    CAST(rv.value AS DATETIME) AS LimiteDerechoInclusive
FROM sys.partitions p
JOIN sys.indexes i ON p.object_id = i.object_id AND p.index_id = i.index_id
JOIN sys.data_spaces ds ON i.data_space_id = ds.data_space_id
JOIN sys.partition_schemes ps ON ds.data_space_id = ps.data_space_id
JOIN sys.partition_functions pf ON ps.function_id = pf.function_id
LEFT JOIN sys.destination_data_spaces dds ON ps.data_space_id = dds.partition_scheme_id AND p.partition_number = dds.destination_id
LEFT JOIN sys.filegroups fg ON dds.data_space_id = fg.data_space_id
LEFT JOIN sys.partition_range_values rv ON pf.function_id = rv.function_id AND p.partition_number = rv.boundary_id
WHERE p.object_id = OBJECT_ID('Sales.SalesOrderDetail_Particionada') AND i.index_id <= 1
ORDER BY p.partition_number;
GO
